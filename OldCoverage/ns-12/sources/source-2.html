92     public void checkAccessControl() throws UnauthorizedAccessException {
93         String userParam = getRequestParamValue(Const.ParamsNames.USER_ID);
94         if (userInfo != null && userParam != null && !userInfo.isAdmin && !userInfo.id.equals(userParam)) {
95             throw new UnauthorizedAccessException("User " + userInfo.id
96                     + " is trying to masquerade as " + userParam + " without admin permission.");
97         }
98 
99         if (authType.getLevel() < getMinAuthLevel().getLevel()) {
100             // Access control level lower than required
101             throw new UnauthorizedAccessException("Not authorized to access this resource.");
102         }
103 
104         if (authType == AuthType.ALL_ACCESS) {
105             // All-access pass granted
106             return;
107         }
108 
109         // All other cases: to be dealt in case-by-case basis
110         checkSpecificAccessControl();
111     }

134     private void initAuthInfo() {
135         if (Config.BACKDOOR_KEY.equals(req.getHeader(Const.HeaderNames.BACKDOOR_KEY))) {
136             authType = AuthType.ALL_ACCESS;
137             userInfo = userProvision.getAdminOnlyUser(getRequestParamValue(Const.ParamsNames.USER_ID));
138             userInfo.isStudent = true;
139             userInfo.isInstructor = true;
140             return;
141         }
142 
143         // The header X-AppEngine-QueueName cannot be spoofed as GAE will strip any user-sent X-AppEngine-QueueName headers.
144         // Reference: https://cloud.google.com/tasks/docs/creating-appengine-handlers#reading_app_engine_task_request_headers
145         String queueNameHeader = req.getHeader("X-AppEngine-QueueName");
146         boolean isRequestFromAppEngineQueue = queueNameHeader != null;
147         if (isRequestFromAppEngineQueue) {
148             userInfo = userProvision.getAdminOnlyUser("AppEngine-" + queueNameHeader);
149         } else {
150             String cookie = HttpRequestHelper.getCookieValueFromRequest(req, Const.SecurityConfig.AUTH_COOKIE_NAME);
151             UserInfoCookie uic = UserInfoCookie.fromCookie(cookie);
152             userInfo = userProvision.getCurrentUser(uic);
153         }
154 
155         authType = userInfo == null ? AuthType.PUBLIC : AuthType.LOGGED_IN;
156 
157         String userParam = getRequestParamValue(Const.ParamsNames.USER_ID);
158         if (userInfo != null && userParam != null && userInfo.isAdmin) {
159             userInfo = userProvision.getMasqueradeUser(userParam);
160             authType = AuthType.MASQUERADE;
161         }
162     }
