153     // a. keep the part of email before "@"
154     //    replace "@" with "."
155     //    replace email host with their first 3 chars. eg, gmail.com -> gma
156     //    append "-demo"

163     //    lebron@gmail.com -> lebron.gma-demo0 // already exists!
164     //    lebron@gmail.com -> lebron.gma-demo1 // already exists!

167     //    lebron@gmail.com -> lebron.gma-demo100 // found! a feasible id

206      *
207      * <p>Here we check the input string is an email or course Id and handle them accordingly;
208      * check the resulting course id, and if bigger than maximumIdLength, cut it so that it equals maximumIdLength.
209      *
210      * @param instructorEmailOrProposedCourseId is the instructor email or a proposed course id that already exists.
211      * @param maximumIdLength is the maximum resulting id length allowed, above which we will cut the part before "@"
212      * @return the proposed course id, e.g.:
213      *         <ul>
214      *         <li>lebron@gmail.com -> lebron.gma-demo</li>
215      *         <li>lebron.gma-demo -> lebron.gma-demo0</li>
216      *         <li>lebron.gma-demo0 -> lebron.gma-demo1</li>
217      *         <li>012345678901234567890123456789.gma-demo9 -> 01234567890123456789012345678.gma-demo10 (being cut)</li>
218      *         </ul>
219      */
220     String generateNextDemoCourseId(String instructorEmailOrProposedCourseId, int maximumIdLength) {
221         boolean isFirstCourseId = instructorEmailOrProposedCourseId.contains("@");
222         if (isFirstCourseId) {
223             return StringHelper.truncateHead(getDemoCourseIdRoot(instructorEmailOrProposedCourseId), maximumIdLength);
224         }
225 
226         boolean isFirstTimeDuplicate = instructorEmailOrProposedCourseId.endsWith("-demo");
227         if (isFirstTimeDuplicate) {
228             return StringHelper.truncateHead(instructorEmailOrProposedCourseId + "0", maximumIdLength);
229         }
230 
231         int lastIndexOfDemo = instructorEmailOrProposedCourseId.lastIndexOf("-demo");
232         String root = instructorEmailOrProposedCourseId.substring(0, lastIndexOfDemo);
233         int previousDedupSuffix = Integer.parseInt(instructorEmailOrProposedCourseId.substring(lastIndexOfDemo + 5));
234 
235         return StringHelper.truncateHead(root + "-demo" + (previousDedupSuffix + 1), maximumIdLength);
236     }
