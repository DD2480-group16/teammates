179     private String cleanSpecialChars(String queryString) {
180         String htmlTagStripPattern = "<[^>]*>";
181 
182         // Solr special characters: + - && || ! ( ) { } [ ] ^ " ~ * ? : \ /
183         String res = queryString.replaceAll(htmlTagStripPattern, "")
184                 .replace("\\", "\\\\")
185                 .replace("+", "\\+")
186                 .replace("-", "\\-")
187                 .replace("&&", "\\&&")
188                 .replace("||", "\\||")
189                 .replace("!", "\\!")
190                 .replace("(", "\\(")
191                 .replace(")", "\\)")
192                 .replace("{", "\\{")
193                 .replace("}", "\\}")
194                 .replace("[", "\\[")
195                 .replace("]", "\\]")
196                 .replace("^", "\\^")
197                 .replace("~", "\\~")
198                 .replace("?", "\\?")
199                 .replace(":", "\\:")
200                 .replace("/", "\\/");
201 
202         // imbalanced double quotes are invalid
203         int count = StringUtils.countMatches(res, "\"");
204         if (count % 2 == 1) {
205             res = res.replace("\"", "");
206         }
207 
208         // use exact match only when there's email-like input
209         if (res.contains("@") && count == 0) {
210             return "\"" + res + "\"";
211         } else {
212             return res;
213         }
