377      * @param commentGiverType type of comment giver
378      */
379     private void verifyIsUserOfCourse(String courseId, String commentGiver, FeedbackParticipantType commentGiverType,
380             boolean isCommentFromFeedbackParticipant) throws EntityDoesNotExistException {
381         if (!isCommentFromFeedbackParticipant) {
382             InstructorAttributes instructor = instructorsLogic.getInstructorForEmail(courseId, commentGiver);
383             if (instructor == null) {
384                 throw new EntityDoesNotExistException("User " + commentGiver
385                         + " is not a registered instructor for course " + courseId + ".");
386             }
387             return;
388         }
389         switch (commentGiverType) {
390         case STUDENTS:
391             StudentAttributes student = studentsLogic.getStudentForEmail(courseId, commentGiver);
392             if (student == null) {
393                 throw new EntityDoesNotExistException("User " + commentGiver + " is not a registered student for course "
394                         + courseId + ".");
395             }
396             break;
397         case INSTRUCTORS:
398             InstructorAttributes instructor = instructorsLogic.getInstructorForEmail(courseId, commentGiver);
399             if (instructor == null) {
400                 throw new EntityDoesNotExistException("User " + commentGiver
401                         + " is not a registered instructor for course " + courseId + ".");
402             }
403             break;
404         case TEAMS:
405             List<String> teams = coursesLogic.getTeamsForCourse(courseId);
406             boolean isTeamPresentInCourse = false;
407             for (String team : teams) {
408                 if (team.equals(commentGiver)) {
409                     isTeamPresentInCourse = true;
410                     break;
411                 }
412             }
413             if (!isTeamPresentInCourse) {
414                 throw new EntityDoesNotExistException("User " + commentGiver + " is not a registered team for course "
415                         + courseId + ".");
416             }
417             break;
418         default:
419             throw new EntityDoesNotExistException("Unknown giver type: " + commentGiverType);
420         }
421     }
