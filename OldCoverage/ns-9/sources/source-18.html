205         StudentAttributes updatedStudent = studentsDb.updateStudent(updateOptions);

207         // cascade email change, if any

212

214         if (isTeamChanged(originalStudent.getTeam(), updatedStudent.getTeam())) {

216                     originalStudent.getTeam(), updatedStudent.getTeam());
217         }
218 
219         // update the new section name in responses
220         if (isSectionChanged(originalStudent.getSection(), updatedStudent.getSection())) {
221             frLogic.updateFeedbackResponsesForChangingSection(updatedStudent.getCourse(), updatedStudent.getEmail(),

314     private String getSectionInvalidityInfo(List<StudentAttributes> mergedList) {
315 
316         StudentAttributes.sortBySectionName(mergedList);
317 
318         List<String> invalidSectionList = new ArrayList<>();
319         int studentsCount = 1;
320         for (int i = 1; i < mergedList.size(); i++) {
321             StudentAttributes currentStudent = mergedList.get(i);
322             StudentAttributes previousStudent = mergedList.get(i - 1);
323             if (currentStudent.getSection().equals(previousStudent.getSection())) {
324                 studentsCount++;
325             } else {
326                 if (studentsCount > Const.SECTION_SIZE_LIMIT) {
327                     invalidSectionList.add(previousStudent.getSection());
328                 }
329                 studentsCount = 1;
330             }
331 
332             if (i == mergedList.size() - 1 && studentsCount > Const.SECTION_SIZE_LIMIT) {
333                 invalidSectionList.add(currentStudent.getSection());
334             }
335         }
336 
337         StringJoiner errorMessage = new StringJoiner(" ");
338         for (String section : invalidSectionList) {
339             errorMessage.add(String.format(
340                     ERROR_ENROLL_EXCEED_SECTION_LIMIT,
341                     Const.SECTION_SIZE_LIMIT, section));
342         }
343 
344         if (!invalidSectionList.isEmpty()) {
345             errorMessage.add(String.format(
346                     ERROR_ENROLL_EXCEED_SECTION_LIMIT_INSTRUCTION,
347                     Const.SECTION_SIZE_LIMIT));
348         }
349 
350         return errorMessage.toString();
351     }

353     private String getTeamInvalidityInfo(List<StudentAttributes> mergedList) {
354         StringJoiner errorMessage = new StringJoiner(" ");
355         StudentAttributes.sortByTeamName(mergedList);
356 
357         List<String> invalidTeamList = new ArrayList<>();
358         for (int i = 1; i < mergedList.size(); i++) {
359             StudentAttributes currentStudent = mergedList.get(i);
360             StudentAttributes previousStudent = mergedList.get(i - 1);
361             if (currentStudent.getTeam().equals(previousStudent.getTeam())
362                     && !currentStudent.getSection().equals(previousStudent.getSection())
363                     && !invalidTeamList.contains(currentStudent.getTeam())) {
364 
365                 errorMessage.add(String.format(ERROR_INVALID_TEAM_NAME,
366                         currentStudent.getTeam(),
367                         previousStudent.getSection(),
368                         currentStudent.getSection()));
369 
370                 invalidTeamList.add(currentStudent.getTeam());
371             }
372         }
373 
374         if (!invalidTeamList.isEmpty()) {
375             errorMessage.add(ERROR_INVALID_TEAM_NAME_INSTRUCTION);
376         }
377 
378         return errorMessage.toString();
379     }
