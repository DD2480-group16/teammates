610             generateOptionsFor = feedbackMsqQuestionDetails.getGenerateOptionsFor();
611         } else {
612             // other question types
613             return;
614         }
615 
616         switch (generateOptionsFor) {
617         case NONE:
618             break;
619         case STUDENTS:
620         case STUDENTS_IN_SAME_SECTION:
621         case STUDENTS_EXCLUDING_SELF:
622             List<StudentAttributes> studentList;
623             if (generateOptionsFor == FeedbackParticipantType.STUDENTS_IN_SAME_SECTION) {
624                 String courseId = feedbackQuestionAttributes.getCourseId();
625                 StudentAttributes studentAttributes =
626                         studentsLogic.getStudentForEmail(emailOfEntityDoingQuestion, courseId);
627                 studentList = studentsLogic.getStudentsForSection(studentAttributes.getSection(), courseId);
628             } else {
629                 studentList = studentsLogic.getStudentsForCourse(feedbackQuestionAttributes.getCourseId());
630             }
631 
632             if (generateOptionsFor == FeedbackParticipantType.STUDENTS_EXCLUDING_SELF) {
633                 studentList.removeIf(studentInList -> studentInList.getEmail().equals(emailOfEntityDoingQuestion));
634             }
635 
636             for (StudentAttributes student : studentList) {
637                 optionList.add(student.getName() + " (" + student.getTeam() + ")");
638             }
639 
640             optionList.sort(null);
641             break;
642         case TEAMS:
643         case TEAMS_IN_SAME_SECTION:
644         case TEAMS_EXCLUDING_SELF:
645             try {
646                 List<String> teams;
647                 if (generateOptionsFor == FeedbackParticipantType.TEAMS_IN_SAME_SECTION) {
648                     String courseId = feedbackQuestionAttributes.getCourseId();
649                     StudentAttributes studentAttributes =
650                             studentsLogic.getStudentForEmail(emailOfEntityDoingQuestion, courseId);
651                     teams = coursesLogic.getTeamsForSection(studentAttributes.getSection(), courseId);
652                 } else {
653                     teams = coursesLogic.getTeamsForCourse(feedbackQuestionAttributes.getCourseId());
654                 }
655 
656                 if (generateOptionsFor == FeedbackParticipantType.TEAMS_EXCLUDING_SELF) {
657                     teams.removeIf(team -> team.equals(teamOfEntityDoingQuestion));
658                 }
659 
660                 for (String team : teams) {
661                     optionList.add(team);
662                 }
663 
664                 optionList.sort(null);
665             } catch (EntityDoesNotExistException e) {
666                 assert false : "Course disappeared";
667             }
668             break;
669         case OWN_TEAM_MEMBERS_INCLUDING_SELF:
670         case OWN_TEAM_MEMBERS:
671             if (teamOfEntityDoingQuestion != null) {
672                 List<StudentAttributes> teamMembers = studentsLogic.getStudentsForTeam(teamOfEntityDoingQuestion,
673                         feedbackQuestionAttributes.getCourseId());
674 
675                 if (generateOptionsFor == FeedbackParticipantType.OWN_TEAM_MEMBERS) {
676                     teamMembers.removeIf(teamMember -> teamMember.getEmail().equals(emailOfEntityDoingQuestion));
677                 }
678 
679                 teamMembers.forEach(teamMember -> optionList.add(teamMember.getName()));
680 
681                 optionList.sort(null);
682             }
683             break;
684         case INSTRUCTORS:
685             List<InstructorAttributes> instructorList =
686                     instructorsLogic.getInstructorsForCourse(feedbackQuestionAttributes.getCourseId());
687 
688             for (InstructorAttributes instructor : instructorList) {
689                 optionList.add(instructor.getName());
690             }
691 
692             optionList.sort(null);
693             break;
694         default:
695             assert false : "Trying to generate options for neither students, teams nor instructors";
696             break;
697         }
698 
699         if (feedbackQuestionAttributes.getQuestionType() == FeedbackQuestionType.MCQ) {
700             FeedbackMcqQuestionDetails feedbackMcqQuestionDetails =
701                     (FeedbackMcqQuestionDetails) feedbackQuestionAttributes.getQuestionDetailsCopy();
702             feedbackMcqQuestionDetails.setMcqChoices(optionList);
703             feedbackQuestionAttributes.setQuestionDetails(feedbackMcqQuestionDetails);
704         } else if (feedbackQuestionAttributes.getQuestionType() == FeedbackQuestionType.MSQ) {
705             FeedbackMsqQuestionDetails feedbackMsqQuestionDetails =
706                     (FeedbackMsqQuestionDetails) feedbackQuestionAttributes.getQuestionDetailsCopy();
707             feedbackMsqQuestionDetails.setMsqChoices(optionList);
708             feedbackQuestionAttributes.setQuestionDetails(feedbackMsqQuestionDetails);
709         }
710     }

758      *
759      * <p>Cascade adjust the question number of questions in the same session.
760      *
761      * <p>Cascade adjust the existing response of the question.
762      *
763      * @return updated feedback question
764      * @throws InvalidParametersException if attributes to update are not valid
765      * @throws EntityDoesNotExistException if the feedback question cannot be found
766      */
767     public FeedbackQuestionAttributes updateFeedbackQuestionCascade(FeedbackQuestionAttributes.UpdateOptions updateOptions)
768             throws InvalidParametersException, EntityDoesNotExistException {
769         FeedbackQuestionAttributes oldQuestion = fqDb.getFeedbackQuestion(updateOptions.getFeedbackQuestionId());
770         if (oldQuestion == null) {
771             throw new EntityDoesNotExistException("Trying to update a feedback question that does not exist.");
772         }
773 
774         FeedbackQuestionAttributes newQuestion = oldQuestion.getCopy();
775         newQuestion.update(updateOptions);
776         int oldQuestionNumber = oldQuestion.getQuestionNumber();
777         int newQuestionNumber = newQuestion.getQuestionNumber();
778 
779         List<FeedbackQuestionAttributes> previousQuestionsInSession = new ArrayList<>();
780         if (oldQuestionNumber != newQuestionNumber) {
781             // get questions in session before update
782             String feedbackSessionName = oldQuestion.getFeedbackSessionName();
783             String courseId = oldQuestion.getCourseId();
784             previousQuestionsInSession = getFeedbackQuestionsForSession(feedbackSessionName, courseId);
785         }
786 
787         // update question
788         FeedbackQuestionAttributes updatedQuestion = fqDb.updateFeedbackQuestion(updateOptions);
789 
790         if (oldQuestionNumber != newQuestionNumber) {
791             // shift other feedback questions (generate an empty "slot")
792             adjustQuestionNumbers(oldQuestionNumber, newQuestionNumber, previousQuestionsInSession);

794

796         if (oldQuestion.areResponseDeletionsRequiredForChanges(updatedQuestion)) {
797             frLogic.deleteFeedbackResponsesForQuestionCascade(oldQuestion.getId());
798         }

803     /**
804      * Adjust questions between the old and new number,
805      * if the new number is smaller, then shift up (increase qn#) all questions in between.
806      * if the new number is bigger, then shift down(decrease qn#) all questions in between.
807      */
808     private void adjustQuestionNumbers(int oldQuestionNumber,
809             int newQuestionNumber, List<FeedbackQuestionAttributes> questions) {
810         try {
811             if (oldQuestionNumber > newQuestionNumber && oldQuestionNumber >= 1) {
812                 for (int i = oldQuestionNumber - 1; i >= newQuestionNumber; i--) {
813                     FeedbackQuestionAttributes question = questions.get(i - 1);
814                     fqDb.updateFeedbackQuestion(
815                             FeedbackQuestionAttributes.updateOptionsBuilder(question.getId())
816                                     .withQuestionNumber(question.getQuestionNumber() + 1)
817                                     .build());
818                 }
819             } else if (oldQuestionNumber < newQuestionNumber && oldQuestionNumber < questions.size()) {
820                 for (int i = oldQuestionNumber + 1; i <= newQuestionNumber; i++) {
821                     FeedbackQuestionAttributes question = questions.get(i - 1);
822                     fqDb.updateFeedbackQuestion(
823                             FeedbackQuestionAttributes.updateOptionsBuilder(question.getId())
824                                     .withQuestionNumber(question.getQuestionNumber() - 1)
825                                     .build());
826                 }
827             }
828         } catch (InvalidParametersException | EntityDoesNotExistException e) {
829             assert false : "Adjusting question number should not cause: " + e.getMessage();
830         }
831     }
