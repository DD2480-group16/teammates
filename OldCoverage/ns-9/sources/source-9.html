233     private boolean isFeedbackParticipantNameVisibleToUser(
234             FeedbackQuestionAttributes question, FeedbackResponseAttributes response,
235             String userEmail, boolean isInstructor, boolean isGiverName, CourseRoster roster) {
236         List<FeedbackParticipantType> showNameTo = isGiverName
237                                                  ? question.getShowGiverNameTo()
238                                                  : question.getShowRecipientNameTo();
239         for (FeedbackParticipantType type : showNameTo) {
240             switch (type) {
241             case INSTRUCTORS:
242                 if (roster.getInstructorForEmail(userEmail) != null && isInstructor) {
243                     return true;
244                 }
245                 break;
246             case OWN_TEAM_MEMBERS:
247             case OWN_TEAM_MEMBERS_INCLUDING_SELF:
248                 // Refers to Giver's Team Members
249                 if (roster.isStudentsInSameTeam(response.getGiver(), userEmail)) {
250                     return true;
251                 }
252                 break;
253             case RECEIVER:
254                 // Response to team
255                 if (question.getRecipientType().isTeam()) {
256                     if (roster.isStudentInTeam(userEmail, response.getRecipient())) {
257                         // this is a team name
258                         return true;
259                     }
260                     break;
261                     // Response to individual
262                 } else if (response.getRecipient().equals(userEmail)) {
263                     return true;
264                 } else {
265                     break;
266                 }
267             case RECEIVER_TEAM_MEMBERS:
268                 // Response to team; recipient = teamName
269                 if (question.getRecipientType().isTeam()) {
270                     if (roster.isStudentInTeam(userEmail, response.getRecipient())) {
271                         // this is a team name
272                         return true;
273                     }
274                     break;
275                 } else if (roster.isStudentsInSameTeam(response.getRecipient(), userEmail)) {
276                     // Response to individual
277                     return true;
278                 }
279                 break;
280             case STUDENTS:
281                 if (roster.isStudentInCourse(userEmail)) {
282                     return true;
283                 }
284                 break;
285             default:
286                 assert false : "Invalid FeedbackParticipantType for showNameTo in "
287                         + "FeedbackResponseLogic.isFeedbackParticipantNameVisibleToUser()";
288                 break;
289             }
290         }
291         return false;
292     }

358         Map<String, List<FeedbackResponseCommentAttributes>> relatedCommentsMap = new HashMap<>();
359         if (isCourseWide) {

361             for (FeedbackQuestionAttributes qn : allQuestions) {

376         Map<String, Boolean> responseRecipientVisibilityTable = new HashMap<>();

381             FeedbackQuestionAttributes correspondingQuestion = allQuestionsMap.get(response.getFeedbackQuestionId());
382             if (correspondingQuestion == null) {
383                 // orphan response without corresponding question, ignore it
384                 continue;

407             FeedbackQuestionAttributes relatedQuestion = relatedQuestionsMap.get(frc.getFeedbackQuestionId());

409             if (relatedQuestion == null || relatedResponse == null) {

422         }

424

427         if (isCourseWide) {

431         }
432         RequestTracer.checkRemainingTime();
433

435                 responseGiverVisibilityTable, responseRecipientVisibilityTable, relatedCommentsMap,

437     }

576                     if (section != null
577                             && !giverInfo.getSectionName().equals(section)
578                             && !recipientInfo.getSectionName().equals(section)) {
579                         continue;
580                     }
581 
582                     FeedbackResponseAttributes missingResponse =
583                             FeedbackResponseAttributes.builder(questionId, giverIdentifier, recipientIdentifier)
584                                     .withCourseId(courseId)
585                                     .withFeedbackSessionName(feedbackSessionName)
586                                     .withGiverSection(giverInfo.getSectionName())
587                                     .withRecipientSection(recipientInfo.getSectionName())
588                                     .withResponseDetails(new FeedbackTextResponseDetails("No Response"))
589                                     .build();
590 
591                     // check visibility of the missing response
592                     boolean isVisibleResponse = isResponseVisibleForUser(
593                             instructor.getEmail(), true, null, Collections.emptySet(),
594                             missingResponse, correspondingQuestion, instructor);
595                     if (!isVisibleResponse) {

603                     responseRecipientVisibilityTable.put(missingResponse.getId(),
604                             isNameVisibleToUser(correspondingQuestion, missingResponse,
605                                     instructor.getEmail(), true, false, courseRoster));

614     private boolean isResponseVisibleForUser(
615             String userEmail, boolean isInstructor, StudentAttributes student,
616             Set<String> studentsEmailInTeam, FeedbackResponseAttributes response,
617             FeedbackQuestionAttributes relatedQuestion, InstructorAttributes instructor) {
618 
619         boolean isVisibleResponse = false;
620         if (isInstructor && relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.INSTRUCTORS)
621                 || response.getRecipient().equals(userEmail)
622                 && relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.RECEIVER)
623                 || response.getGiver().equals(userEmail)
624                 || !isInstructor && relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.STUDENTS)) {

626         } else if (studentsEmailInTeam != null && !isInstructor) {

628                     && relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.RECEIVER)
629                     && response.getRecipient().equals(student.getTeam())) {

632                     && relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.RECEIVER)
633                     && response.getRecipient().equals(student.getTeam())) {
634                 isVisibleResponse = true;
635             } else if (relatedQuestion.getGiverType() == FeedbackParticipantType.TEAMS
636                     && response.getGiver().equals(student.getTeam())) {
637                 isVisibleResponse = true;
638             } else if (relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.OWN_TEAM_MEMBERS)
639                     && studentsEmailInTeam.contains(response.getGiver())) {
640                 isVisibleResponse = true;
641             } else if (relatedQuestion.isResponseVisibleTo(FeedbackParticipantType.RECEIVER_TEAM_MEMBERS)
642                     && studentsEmailInTeam.contains(response.getRecipient())) {
643                 isVisibleResponse = true;
644             }
645         }
646         if (isVisibleResponse && instructor != null) {
647             boolean isGiverSectionRestricted =
648                     !instructor.isAllowedForPrivilege(response.getGiverSection(),
649                             response.getFeedbackSessionName(),
650                             Const.InstructorPermissions.CAN_VIEW_SESSION_IN_SECTIONS);
651             // If instructors are not restricted to view the giver's section,
652             // they are allowed to view responses to GENERAL, subject to visibility options
653             boolean isRecipientSectionRestricted =
654                     relatedQuestion.getRecipientType() != FeedbackParticipantType.NONE
655                             && !instructor.isAllowedForPrivilege(response.getRecipientSection(),
656                             response.getFeedbackSessionName(),
657                             Const.InstructorPermissions.CAN_VIEW_SESSION_IN_SECTIONS);
658 
659             boolean isNotAllowedForInstructor = isGiverSectionRestricted || isRecipientSectionRestricted;
660             if (isNotAllowedForInstructor) {
661                 isVisibleResponse = false;
662             }
663         }
664         return isVisibleResponse;
665     }

678      * (e.g. associated response ID, giverSection and recipientSection).
679      *
680      * <p>If the giver/recipient field is changed, the response is updated by recreating the response
681      * as question-giver-recipient is the primary key.
682      *
683      * @return updated feedback response
684      * @throws InvalidParametersException if attributes to update are not valid
685      * @throws EntityDoesNotExistException if the comment cannot be found
686      * @throws EntityAlreadyExistsException if the response cannot be updated
687      *         by recreation because of an existent response
688      */
689     public FeedbackResponseAttributes updateFeedbackResponseCascade(FeedbackResponseAttributes.UpdateOptions updateOptions)
690             throws InvalidParametersException, EntityDoesNotExistException, EntityAlreadyExistsException {
691 
692         FeedbackResponseAttributes oldResponse = frDb.getFeedbackResponse(updateOptions.getFeedbackResponseId());
693         FeedbackResponseAttributes newResponse = frDb.updateFeedbackResponse(updateOptions);
694 
695         boolean isResponseIdChanged = !oldResponse.getId().equals(newResponse.getId());
696         boolean isGiverSectionChanged = !oldResponse.getGiverSection().equals(newResponse.getGiverSection());
697         boolean isRecipientSectionChanged = !oldResponse.getRecipientSection().equals(newResponse.getRecipientSection());
698 
699         if (isResponseIdChanged || isGiverSectionChanged || isRecipientSectionChanged) {
700             List<FeedbackResponseCommentAttributes> responseComments =
701                     frcLogic.getFeedbackResponseCommentForResponse(oldResponse.getId());
702             for (FeedbackResponseCommentAttributes responseComment : responseComments) {
703                 FeedbackResponseCommentAttributes.UpdateOptions.Builder updateOptionsBuilder =
704                         FeedbackResponseCommentAttributes.updateOptionsBuilder(responseComment.getId());
705 
706                 if (isResponseIdChanged) {
707                     updateOptionsBuilder.withFeedbackResponseId(newResponse.getId());
708                 }
709 
710                 if (isGiverSectionChanged) {
711                     updateOptionsBuilder.withGiverSection(newResponse.getGiverSection());
712                 }
713 
714                 if (isRecipientSectionChanged) {
715                     updateOptionsBuilder.withReceiverSection(newResponse.getRecipientSection());
716                 }
717 
718                 frcLogic.updateFeedbackResponseComment(updateOptionsBuilder.build());
719             }
720         }
721 
722         return newResponse;
723     }
