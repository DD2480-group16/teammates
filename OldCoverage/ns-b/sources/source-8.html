213     /**
214      * Updates a feedback response with {@link FeedbackResponseAttributes.UpdateOptions}.
215      *
216      * <p>If the giver/recipient field is changed, the response is updated by recreating the response
217      * as question-giver-recipient is the primary key.
218      *
219      * @return updated feedback response
220      * @throws InvalidParametersException if attributes to update are not valid
221      * @throws EntityDoesNotExistException if the comment cannot be found
222      * @throws EntityAlreadyExistsException if the response cannot be updated
223      *         by recreation because of an existent response
224      */
225     public FeedbackResponseAttributes updateFeedbackResponse(FeedbackResponseAttributes.UpdateOptions updateOptions)
226             throws EntityDoesNotExistException, InvalidParametersException, EntityAlreadyExistsException {
227         assert updateOptions != null;
228 
229         FeedbackResponse oldResponse = getFeedbackResponseEntity(updateOptions.getFeedbackResponseId());
230         if (oldResponse == null) {
231             throw new EntityDoesNotExistException(ERROR_UPDATE_NON_EXISTENT);
232         }
233 
234         FeedbackResponseAttributes newAttributes = makeAttributes(oldResponse);
235         newAttributes.update(updateOptions);
236 
237         newAttributes.sanitizeForSaving();
238         if (!newAttributes.isValid()) {
239             throw new InvalidParametersException(newAttributes.getInvalidityInfo());
240         }
241 
242         if (newAttributes.getRecipient().equals(oldResponse.getRecipientEmail())
243                 && newAttributes.getGiver().equals(oldResponse.getGiverEmail())) {
244 
245             // update only if change
246             boolean hasSameAttributes =
247                     this.<String>hasSameValue(oldResponse.getGiverSection(), newAttributes.getGiverSection())
248                     && this.<String>hasSameValue(oldResponse.getRecipientSection(), newAttributes.getRecipientSection())
249                     && this.<String>hasSameValue(
250                             oldResponse.getAnswer(), newAttributes.getSerializedFeedbackResponseDetail());
251             if (hasSameAttributes) {
252                 log.info(String.format(
253                         OPTIMIZED_SAVING_POLICY_APPLIED, FeedbackResponse.class.getSimpleName(), updateOptions));
254                 return newAttributes;
255             }
256 
257             oldResponse.setGiverSection(newAttributes.getGiverSection());
258             oldResponse.setRecipientSection(newAttributes.getRecipientSection());
259             oldResponse.setAnswer(newAttributes.getSerializedFeedbackResponseDetail());
260 
261             saveEntity(oldResponse);
262 
263             return makeAttributes(oldResponse);
264         } else {
265             // need to recreate the entity
266             newAttributes = FeedbackResponseAttributes
267                     .builder(newAttributes.getFeedbackQuestionId(), newAttributes.getGiver(),
268                              newAttributes.getRecipient())
269                     .withCourseId(newAttributes.getCourseId())
270                     .withFeedbackSessionName(newAttributes.getFeedbackSessionName())
271                     .withResponseDetails(newAttributes.getResponseDetailsCopy())
272                     .withGiverSection(newAttributes.getGiverSection())
273                     .withRecipientSection(newAttributes.getRecipientSection())
274                     .build();
275             newAttributes = createEntity(newAttributes);
276             deleteEntity(Key.create(FeedbackResponse.class, oldResponse.getId()));
277 
278             return newAttributes;
279         }
280     }
