454     private List<EmailWrapper> generateFeedbackSessionClosingOrClosedEmails(

458         boolean isEmailNeededForStudents = fsLogic.isFeedbackSessionForUserTypeToAnswer(session, false);
459         boolean isEmailNeededForInstructors = fsLogic.isFeedbackSessionForUserTypeToAnswer(session, true);
460 
461         if (isEmailNeededForStudents) {

463

471                             + "session " + session.getFeedbackSessionName() + " does not exist");
472                     // Course or session cannot be found for one student => it will be the case for all students
473                     // Do not waste time looping through all students
474                     break;
475                 }
476             }
477         }
478 
479         if (isEmailNeededForInstructors) {
480             List<InstructorAttributes> instructorsForCourse =
481                     instructorsLogic.getInstructorsForCourse(session.getCourseId());
482 
483             for (InstructorAttributes instructor : instructorsForCourse) {
484                 try {
485                     if (!fsLogic.isFeedbackSessionAttemptedByUser(session, instructor.getEmail(), true)) {
486                         instructors.add(instructor);
487                     }
488                 } catch (EntityDoesNotExistException e) {
489                     log.severe("Course " + session.getCourseId() + " does not exist or "
490                             + "session " + session.getFeedbackSessionName() + " does not exist");
491                     // Course or session cannot be found for one instructor => it will be the case for all instructors
492                     // Do not waste time looping through all instructors
493                     break;
494                 }
495             }
496         }
497 
498         String status;

500         if (emailType == EmailType.FEEDBACK_CLOSED) {
501             status = FEEDBACK_STATUS_SESSION_CLOSED;
502             action = FEEDBACK_ACTION_VIEW;
503         } else {

505             action = FEEDBACK_ACTION_SUBMIT_EDIT_OR_VIEW;
506         }

510         List<InstructorAttributes> instructorsToNotify = isEmailNeededForStudents
511                 ? instructorsLogic.getCoOwnersForCourse(session.getCourseId())
512                 : new ArrayList<>();
513         return generateFeedbackSessionEmailBases(course, session, students, instructors, instructorsToNotify, template,
514                 emailType, action);

578     private List<EmailWrapper> generateFeedbackSessionEmailBases(
579             CourseAttributes course, FeedbackSessionAttributes session, List<StudentAttributes> students,
580             List<InstructorAttributes> instructors, List<InstructorAttributes> instructorsToNotify, String template,
581             EmailType type, String feedbackAction) {
582         StringBuilder studentAdditionalContactBuilder = new StringBuilder();
583         StringBuilder instructorAdditionalContactBuilder = new StringBuilder();
584         if (type == EmailType.FEEDBACK_CLOSING || type == EmailType.FEEDBACK_SESSION_REMINDER) {

586             instructorAdditionalContactBuilder.append(HTML_NO_ACTION_REQUIRED);

589         instructorAdditionalContactBuilder.append(getAdditionalContactInformationFragment(course, true));
590 
591         List<EmailWrapper> emails = new ArrayList<>();
592         for (StudentAttributes student : students) {
593             emails.add(generateFeedbackSessionEmailBaseForStudents(course, session, student,
594                     template, type, feedbackAction, studentAdditionalContactBuilder.toString()));
595         }
596         for (InstructorAttributes instructor : instructors) {
597             emails.add(generateFeedbackSessionEmailBaseForInstructors(course, session, instructor,
598                     template, type, feedbackAction, instructorAdditionalContactBuilder.toString()));
599         }
600         for (InstructorAttributes instructor : instructorsToNotify) {
601             emails.add(generateFeedbackSessionEmailBaseForNotifiedInstructors(course, session, instructor,
602                     template, type, feedbackAction, studentAdditionalContactBuilder.toString()));
603         }
