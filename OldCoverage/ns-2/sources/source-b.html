47     private Map<String, List<FeedbackResponseAttributes>> buildQuestionToResponseMap(
48             List<FeedbackResponseAttributes> responses) {
49         // build question to response map
50         Map<String, List<FeedbackResponseAttributes>> questionToResponseMap = new LinkedHashMap<>();
51         List<FeedbackQuestionAttributes> questions = new ArrayList<>(questionsMap.values());
52         for (FeedbackQuestionAttributes question : questions) {
