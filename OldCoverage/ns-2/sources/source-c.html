349     static String pointsToString(double[][] array) {
350         StringBuilder returnValue = new StringBuilder();
351         boolean isSquareArray = array.length == array[0].length;
352         int teamSize = (array.length - 1) / 3;
353         int firstDividerLocation = teamSize - 1;
354         int secondDividerLocation = teamSize * 2 - 1;
355         int thirdDividerLocation = secondDividerLocation + 1;
356         for (int i = 0; i < array.length; i++) {
357             returnValue.append(Arrays.toString(array[i])).append(System.lineSeparator());
358             if (isSquareArray) {
359                 continue;
360             }
361             if (i == firstDividerLocation || i == secondDividerLocation || i == thirdDividerLocation) {
362                 returnValue.append("=======================")

364             }
