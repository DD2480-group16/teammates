191             Map<String, List<String>> teamMembersEmail,
192             Map<String, List<FeedbackResponseAttributes>> teamResponses) {
193         Map<String, int[][]> teamSubmissionArray = new LinkedHashMap<>();
194         for (String team : teamNames) {
195             int teamSize = teamMembersEmail.get(team).size();
196             teamSubmissionArray.put(team, new int[teamSize][teamSize]);
197             //Initialize all as not submitted.
198             for (int i = 0; i < teamSize; i++) {
199                 for (int j = 0; j < teamSize; j++) {
200                     teamSubmissionArray.get(team)[i][j] = Const.POINTS_NOT_SUBMITTED;
201                 }
202             }
203             //Fill in submitted points
204             List<FeedbackResponseAttributes> teamResponseList = teamResponses.get(team);
205             List<String> memberEmailList = teamMembersEmail.get(team);
206             for (FeedbackResponseAttributes response : teamResponseList) {
207                 int giverIndx = memberEmailList.indexOf(response.getGiver());
208                 int recipientIndx = memberEmailList.indexOf(response.getRecipient());
209                 if (giverIndx == -1 || recipientIndx == -1) {
210                     continue;
211                 }
212                 int points = ((FeedbackContributionResponseDetails) response.getResponseDetailsCopy()).getAnswer();
213                 teamSubmissionArray.get(team)[giverIndx][recipientIndx] = points;
214             }
215         }
216         return teamSubmissionArray;

264

267             boolean validAnswer = false;
268 
269             // Valid answers: 0, 10, 20, .... 190, 200
270             boolean isValidRange = details.getAnswer() >= 0 && details.getAnswer() <= 200;
271             boolean isMultipleOf10 = details.getAnswer() % 10 == 0;
272 
273             if (isValidRange && isMultipleOf10) {
274                 validAnswer = true;
275             }
276             if (details.getAnswer() == Const.POINTS_NOT_SURE && isNotSureAllowed

278                 validAnswer = true;
279             }
280             if (!validAnswer) {
281                 errors.add(CONTRIB_ERROR_INVALID_OPTION);
282             }
283         }
284         return errors;
