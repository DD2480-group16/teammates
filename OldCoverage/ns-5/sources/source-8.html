113     @Override
114     public List<String> validateQuestionDetails() {
115         List<String> errors = new ArrayList<>();
116         if (generateOptionsFor == FeedbackParticipantType.NONE) {
117 
118             if (msqChoices.size() < MSQ_MIN_NUM_OF_CHOICES) {
119                 errors.add(MSQ_ERROR_NOT_ENOUGH_CHOICES
120                            + MSQ_MIN_NUM_OF_CHOICES + ".");
121             }
122 
123             // If there are Empty Msq options entered trigger this error
124             boolean isEmptyMsqOptionEntered = msqChoices.stream().anyMatch(msqText -> msqText.trim().equals(""));
125             if (isEmptyMsqOptionEntered) {
126                 errors.add(MSQ_ERROR_EMPTY_MSQ_OPTION);
127             }
128 
129             // If weights are enabled, number of choices and weights should be same.
130             // If a user enters an invalid weight for a valid choice,
131             // the msqChoices.size() will be greater than msqWeights.size(), in that case
132             // trigger this error.
133             if (hasAssignedWeights && msqChoices.size() != msqWeights.size()) {
134                 errors.add(MSQ_ERROR_INVALID_WEIGHT);
135             }
136 
137             // If weights are not enabled, but weight list is not empty or otherWeight is not 0
138             // In that case, trigger this error.
139             if (!hasAssignedWeights && (!msqWeights.isEmpty() || msqOtherWeight != 0)) {
140                 errors.add(MSQ_ERROR_INVALID_WEIGHT);
141             }
142 
143             // If weight is enabled, but other option is disabled, and msqOtherWeight is not 0
144             // In that case, trigger this error.
145             if (hasAssignedWeights && !otherEnabled && msqOtherWeight != 0) {
146                 errors.add(MSQ_ERROR_INVALID_WEIGHT);
147             }
148 
149             // If weights are negative, trigger this error.
150             if (hasAssignedWeights && !msqWeights.isEmpty()) {
151                 msqWeights.stream()
152                         .filter(weight -> weight < 0)
153                         .forEach(weight -> errors.add(MSQ_ERROR_INVALID_WEIGHT));
154             }
155 
156             // If 'Other' option is enabled, and other weight has negative value,
157             // trigger this error.
158             if (hasAssignedWeights && otherEnabled && msqOtherWeight < 0) {
159                 errors.add(MSQ_ERROR_INVALID_WEIGHT);
160             }
161 
162             //If there are duplicate mcq options trigger this error
163             boolean isDuplicateOptionsEntered = msqChoices.stream().map(String::trim).distinct().count()
164                                                 != msqChoices.size();
165             if (isDuplicateOptionsEntered) {
166                 errors.add(MSQ_ERROR_DUPLICATE_MSQ_OPTION);
167             }
168         }
169 
170         boolean isMaxSelectableChoicesEnabled = maxSelectableChoices != Const.POINTS_NO_VALUE;
171         boolean isMinSelectableChoicesEnabled = minSelectableChoices != Const.POINTS_NO_VALUE;
172         boolean isMsqChoiceValidatable = generateOptionsFor == FeedbackParticipantType.NONE;
173 
174         int numOfMsqChoices = msqChoices.size() + (otherEnabled ? 1 : 0);
175         if (isMsqChoiceValidatable && isMaxSelectableChoicesEnabled) {
176             if (numOfMsqChoices < maxSelectableChoices) {
177                 errors.add(MSQ_ERROR_MAX_SELECTABLE_EXCEEDED_TOTAL);
178             } else if (maxSelectableChoices < 2) {
179                 errors.add(MSQ_ERROR_MIN_FOR_MAX_SELECTABLE_CHOICES);
180             }
181         }
182 
183         if (isMsqChoiceValidatable && isMinSelectableChoicesEnabled) {
184             if (minSelectableChoices < 1) {
185                 errors.add(MSQ_ERROR_MIN_FOR_MIN_SELECTABLE_CHOICES);
186             }
187             if (minSelectableChoices > numOfMsqChoices) {
188                 errors.add(MSQ_ERROR_MIN_SELECTABLE_MORE_THAN_NUM_CHOICES);
189             }
190         }
191 
192         if (isMaxSelectableChoicesEnabled && isMinSelectableChoicesEnabled
193                 && minSelectableChoices > maxSelectableChoices) {
194             errors.add(MSQ_ERROR_MIN_SELECTABLE_EXCEEDED_MAX_SELECTABLE);
195         }
196 
197         return errors;
198     }

201     public List<String> validateResponsesDetails(List<FeedbackResponseDetails> responses, int numRecipients) {
202         List<String> errors = new ArrayList<>();
203 
204         for (FeedbackResponseDetails response : responses) {
205             FeedbackMsqResponseDetails details = (FeedbackMsqResponseDetails) response;
206 
207             // number of Msq options selected including other option
208             int totalChoicesSelected = details.getAnswers().size();
209             boolean isMaxSelectableEnabled = maxSelectableChoices != Const.POINTS_NO_VALUE;
210             boolean isMinSelectableEnabled = minSelectableChoices != Const.POINTS_NO_VALUE;
211             boolean isNoneOfTheAboveOptionEnabled =
212                     details.getAnswers().contains(MSQ_ANSWER_NONE_OF_THE_ABOVE);
213 
214             // if other is not enabled and other is selected as an answer trigger this error
215             if (details.isOther() && !otherEnabled) {
216                 errors.add(MSQ_ERROR_INVALID_OPTION);
217             }
218 
219             // if other is not chosen while other field is not empty trigger this error
220             if (otherEnabled && !details.isOther() && !details.getOtherFieldContent().isEmpty()) {
221                 errors.add(MSQ_ERROR_INVALID_OPTION);
222             }
223 
224             List<String> validChoices = new ArrayList<>(msqChoices);
225             if (otherEnabled && details.isOther()) {
226                 // other field content becomes a valid choice if other is enabled
227                 validChoices.add(details.getOtherFieldContent());
228             }
229             // if selected answers are not a part of the Msq option list trigger this error
230             boolean isAnswersPartOfChoices = validChoices.containsAll(details.getAnswers());
231             if (!isAnswersPartOfChoices && !isNoneOfTheAboveOptionEnabled) {
232                 errors.add(details.getAnswerString() + " " + MSQ_ERROR_INVALID_OPTION);
233             }
234 
235             // if other option is selected but no text is provided trigger this error
236             if (details.isOther() && "".equals(details.getOtherFieldContent().trim())) {
237                 errors.add(MSQ_ERROR_OTHER_CONTENT_NOT_PROVIDED);
238             }
239 
240             // if other option is selected but not in the answer list trigger this error
241             if (details.isOther() && !details.getAnswers().contains(details.getOtherFieldContent())) {
242                 errors.add(MSQ_ERROR_OTHER_CONTENT_NOT_PROVIDED);
243             }
244 
245             // if total choices selected exceed maximum choices allowed trigger this error
246             if (isMaxSelectableEnabled && totalChoicesSelected > maxSelectableChoices) {
247                 errors.add(MSQ_ERROR_NUM_SELECTED_MORE_THAN_MAXIMUM + maxSelectableChoices);
248             }
249 
250             if (isMinSelectableEnabled) {
251                 // if total choices selected is less than the minimum required choices
252                 if (totalChoicesSelected < minSelectableChoices) {
253                     errors.add(MSQ_ERROR_NUM_SELECTED_LESS_THAN_MINIMUM + minSelectableChoices);
254                 }
255                 // if minimumSelectableChoices is enabled and None of the Above is selected as an answer trigger this error
256                 if (isNoneOfTheAboveOptionEnabled) {
257                     errors.add(MSQ_ERROR_INVALID_OPTION);
258                 }
259             } else {
260                 // if none of the above is selected AND other options are selected trigger this error
261                 if ((details.getAnswers().size() > 1 || details.isOther()) && isNoneOfTheAboveOptionEnabled) {
262                     errors.add(MSQ_ERROR_NONE_OF_THE_ABOVE_ANSWER);
263                 }
264             }
265         }
266         return errors;
267     }
