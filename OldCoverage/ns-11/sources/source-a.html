56     @SuppressWarnings("PMD.AvoidCatchingThrowable") // used as fallback
57     private void invokeServlet(HttpServletRequest req, HttpServletResponse resp) throws IOException {
58         int statusCode = 0;
59         Action action = null;
60         try {
61             action = ActionFactory.getAction(req, req.getMethod());
62             action.init(req);
63             action.checkAccessControl();
64 
65             ActionResult result = action.execute();
66             statusCode = result.getStatusCode();
67             result.send(resp);
68         } catch (ActionMappingException e) {
69             statusCode = e.getStatusCode();
70             throwErrorBasedOnRequester(req, resp, e, statusCode);
71         } catch (InvalidHttpRequestBodyException | InvalidHttpParameterException e) {
72             statusCode = HttpStatus.SC_BAD_REQUEST;
73             throwErrorBasedOnRequester(req, resp, e, statusCode);
74         } catch (UnauthorizedAccessException uae) {
75             statusCode = HttpStatus.SC_FORBIDDEN;
76             log.warning(uae.getClass().getSimpleName() + " caught by WebApiServlet: " + uae.getMessage(), uae);
77             throwError(resp, statusCode,
78                     uae.isShowErrorMessage() ? uae.getMessage() : "You are not authorized to access this resource.");
79         } catch (EntityNotFoundException enfe) {
80             statusCode = HttpStatus.SC_NOT_FOUND;
81             log.warning(enfe.getClass().getSimpleName() + " caught by WebApiServlet: " + enfe.getMessage(), enfe);
82             throwError(resp, statusCode, enfe.getMessage());
83         } catch (InvalidOperationException ioe) {
84             statusCode = HttpStatus.SC_CONFLICT;
85             log.warning(ioe.getClass().getSimpleName() + " caught by WebApiServlet: " + ioe.getMessage(), ioe);
86             throwError(resp, statusCode, ioe.getMessage());
87         } catch (DeadlineExceededException dee) {
88             statusCode = HttpStatus.SC_GATEWAY_TIMEOUT;
89             log.severe(dee.getClass().getSimpleName() + " caught by WebApiServlet", dee);
90             throwError(resp, statusCode, "The request exceeded the server timeout limit. Please try again later.");
91         } catch (DatastoreException e) {
92             statusCode = HttpStatus.SC_INTERNAL_SERVER_ERROR;
93             log.severe(e.getClass().getSimpleName() + " caught by WebApiServlet: " + e.getMessage(), e);
94             throwError(resp, statusCode, e.getMessage());
95         } catch (Throwable t) {
96             statusCode = HttpStatus.SC_INTERNAL_SERVER_ERROR;
97             log.severe(t.getClass().getSimpleName() + " caught by WebApiServlet: " + t.getMessage(), t);
98             throwError(resp, statusCode,
99                     "The server encountered an error when processing your request.");
100         } finally {
101             RequestLogUser userInfo = new RequestLogUser();
102             String requestBody = null;
103             String actionClass = null;
104             if (action != null) {
105                 if (action.hasDefinedRequestBody()) {
106                     requestBody = action.getRequestBody();
107                 }
108                 actionClass = action.getClass().getSimpleName();
109                 userInfo = action.getUserInfoForLogging();
110             }
111 
112             log.request(req, statusCode, actionClass, userInfo, requestBody, actionClass);
113         }
114     }

116     private void throwErrorBasedOnRequester(HttpServletRequest req, HttpServletResponse resp, Exception e, int statusCode)
117             throws IOException {
118         // The header X-AppEngine-QueueName cannot be spoofed as GAE will strip any user-sent X-AppEngine-QueueName headers.
119         // Reference: https://cloud.google.com/tasks/docs/creating-appengine-handlers#reading_app_engine_task_request_headers
120         boolean isRequestFromAppEngineQueue = req.getHeader("X-AppEngine-QueueName") != null;
121 
122         if (isRequestFromAppEngineQueue) {
123             log.severe(e.getClass().getSimpleName() + " caught by WebApiServlet: " + e.getMessage(), e);
124 
125             // Response status is not set to 4XX to 5XX to prevent Cloud Tasks retry mechanism because
126             // if the cause of the exception is improper request URL, no amount of retry is going to help.
127             // The action will be inaccurately marked as "success", but the severe log can be used
128             // to trace the origin of the problem.
129             throwError(resp, HttpStatus.SC_ACCEPTED, e.getMessage());
130         } else {
131             log.warning(e.getClass().getSimpleName() + " caught by WebApiServlet: " + e.getMessage(), e);
132             throwError(resp, statusCode, e.getMessage());
133         }
134     }
