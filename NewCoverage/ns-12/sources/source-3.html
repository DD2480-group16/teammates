181     private static Action getAction(String uri, String method) throws ActionMappingException {
182         if (!ACTION_MAPPINGS.containsKey(uri)) {
183             throw new ActionMappingException("Resource with URI " + uri + " is not found.", HttpStatus.SC_NOT_FOUND);
184         }
185 
186         Class<? extends Action> controllerClass =
187                 ACTION_MAPPINGS.getOrDefault(uri, new HashMap<>()).get(method);
188 
189         if (controllerClass == null) {
190             throw new ActionMappingException("Method [" + method + "] is not allowed for URI " + uri + ".",
191                     HttpStatus.SC_METHOD_NOT_ALLOWED);
192         }
193 
194         try {
195             return controllerClass.newInstance();
196         } catch (Exception e) {
197             assert false : "Could not create the action for " + uri;
198             return null;
199         }
200     }
