214     public boolean isClosingWithinTimeLimit(long hours) {
215         Instant now = Instant.now();
216         Duration difference = Duration.between(now, endTime);
217         // If now and start are almost similar, it means the feedback session
218         // is open for only 24 hours.
219         // Hence we do not send a reminder e-mail for feedback session.
220         return now.isAfter(startTime)
221                && difference.compareTo(Duration.ofHours(hours - 1)) >= 0
222                && difference.compareTo(Duration.ofHours(hours)) < 0;
223     }
