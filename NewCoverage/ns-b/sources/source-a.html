66      */
67     public InstructorAttributes regenerateEntityKey(InstructorAttributes originalInstructor)
68             throws EntityAlreadyExistsException {
69         int numTries = 0;
70         while (numTries < MAX_KEY_REGENERATION_TRIES) {
71             Instructor updatedEntity = convertToEntityForSaving(originalInstructor);
72             if (!updatedEntity.getRegistrationKey().equals(originalInstructor.getKey())) {
73                 saveEntity(updatedEntity);
74                 return makeAttributes(updatedEntity);
75             }
76             numTries++;
77         }
78         log.severe("Failed to generate new registration key for instructor after " + MAX_KEY_REGENERATION_TRIES + " tries");
79         throw new EntityAlreadyExistsException("Could not regenerate a new course registration key for the instructor.");
80     }

403         int numTries = 0;

406             Key<Instructor> existingInstructor =
