50         List<FeedbackSession> endEntities = load()
51                 .filter("endTime >", rangeStart)
52                 .filter("endTime <",
53                         Instant.ofEpochMilli(rangeEnd.toEpochMilli()).plus(Const.FEEDBACK_SESSIONS_SEARCH_WINDOW))
54                 .list();
55 
56         List<FeedbackSession> startEntities = load()
57                 .filter("startTime <", rangeEnd)
58                 .filter("startTime >",
59                         Instant.ofEpochMilli(rangeStart.toEpochMilli()).minus(Const.FEEDBACK_SESSIONS_SEARCH_WINDOW))
60                 .list();
61 
62         List<String> startEntitiesIds = startEntities.stream()
63                 .map(session -> session.getCourseId() + "::" + session.getFeedbackSessionName())
64                 .collect(Collectors.toList());
65 
66         List<FeedbackSession> ongoingSessions = endEntities.stream()
67                 .filter(session -> {
68                     String id = session.getCourseId() + "::" + session.getFeedbackSessionName();
69                     return startEntitiesIds.contains(id);
70                 })
71                 .collect(Collectors.toList());
72
