113         throw new EntityAlreadyExistsException("Could not regenerate a new course registration key for the student.");
114     }

216      * Updates a student by {@link StudentAttributes.UpdateOptions}.
217      *
218      * <p>If the student's email is changed, the student is re-created.
219      *
220      * @return updated student
221      * @throws InvalidParametersException if attributes to update are not valid
222      * @throws EntityDoesNotExistException if the student cannot be found
223      * @throws EntityAlreadyExistsException if the student cannot be updated
224      *         by recreation because of an existent student
225      */
226     public StudentAttributes updateStudent(StudentAttributes.UpdateOptions updateOptions)
227             throws EntityDoesNotExistException, InvalidParametersException, EntityAlreadyExistsException {
228         assert updateOptions != null;
229 
230         CourseStudent student = getCourseStudentEntityForEmail(updateOptions.getCourseId(), updateOptions.getEmail());
231         if (student == null) {
232             throw new EntityDoesNotExistException(ERROR_UPDATE_NON_EXISTENT + updateOptions);
233         }
234 
235         StudentAttributes newAttributes = makeAttributes(student);
236         newAttributes.update(updateOptions);
237 
238         newAttributes.sanitizeForSaving();
239         if (!newAttributes.isValid()) {
240             throw new InvalidParametersException(newAttributes.getInvalidityInfo());
241         }
242 
243         boolean isEmailChanged = !student.getEmail().equals(newAttributes.getEmail());
244 
245         if (isEmailChanged) {
246             newAttributes = createEntity(newAttributes);
247             // delete the old student
248             deleteStudent(student.getCourseId(), student.getEmail());
249 
250             return newAttributes;
251         } else {
252             // update only if change
253             boolean hasSameAttributes =
254                     this.<String>hasSameValue(student.getName(), newAttributes.getName())
255                     && this.<String>hasSameValue(student.getComments(), newAttributes.getComments())
256                     && this.<String>hasSameValue(student.getGoogleId(), newAttributes.getGoogleId())
257                     && this.<String>hasSameValue(student.getTeamName(), newAttributes.getTeam())
258                     && this.<String>hasSameValue(student.getSectionName(), newAttributes.getSection());
259             if (hasSameAttributes) {
260                 log.info(String.format(OPTIMIZED_SAVING_POLICY_APPLIED, CourseStudent.class.getSimpleName(), updateOptions));
261                 return newAttributes;
262             }
263 
264             student.setName(newAttributes.getName());
265             student.setComments(newAttributes.getComments());
266             student.setGoogleId(newAttributes.getGoogleId());
267             student.setTeamName(newAttributes.getTeam());
268             student.setSectionName(newAttributes.getSection());
269 
270             saveEntity(student);
271 
272             return makeAttributes(student);
273         }
274     }

384         int numTries = 0;

386             CourseStudent student = attributes.toEntity();
387             Key<CourseStudent> existingStudent =
