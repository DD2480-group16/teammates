175      * Returns true if the server is configured to be the dev server.
176      */
177     public static boolean isDevServer() {
178         // In production server, GAE sets some non-overrideable environment variables.
179         // We will make use of some of them to determine whether the server is dev server or not.
180         // This means that any developer can replicate this condition in dev server,
181         // but it is their own choice and risk should they choose to do so.
182 
183         String version = System.getenv("GAE_VERSION");
184         if (!APP_VERSION.equals(version)) {
185             return true;
186         }
187 
188         String env = System.getenv("GAE_ENV");
189         if ("standard".equals(env)) {
190             // GAE standard
191             String appName = System.getenv("GAE_APPLICATION");
192             return appName == null || !appName.endsWith(APP_ID);
193         }
194 
195         // GAE flexible; GAE_ENV variable should not exist in GAE flexible environment
196         return env != null;
197     }
