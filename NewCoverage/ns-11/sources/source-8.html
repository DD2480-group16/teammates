70             // Can bypass CSRF check with the correct key

112     /**
113      * Validates the HTTP referrer against the request URL.
114      * The origin is the base URL of the HTTP referrer, which includes the protocol and authority
115      * (host name + port number if specified).
116      * Similarly, the target is the base URL of the requested action URL.
117      * For the referrer to be considered valid, origin and target must match exactly.
118      * Otherwise, the request is likely to be a CSRF attack, and is considered invalid.
119      *
120      * <p>Example of malicious request originating from embedded image in email:
121      * <pre>
122      * Request URL: https://teammatesv4.appspot.com/page/instructorCourseDelete?courseid=abcdef
123      * Referrer:    https://mail.google.com/mail/u/0/
124      *
125      * Target: https://teammatesv4.appspot.com
126      * Origin: https://mail.google.com
127      * </pre>
128      * Origin does not match target. This request is invalid.</p>
129      *
130      * <p>Example of legitimate request originating from instructor courses page:
131      * <pre>
132      * Request URL: https://teammatesv4.appspot.com/page/instructorCourseDelete?courseid=abcdef
133      * Referrer:    https://teammatesv4.appspot.com/page/instructorCoursesPage
134      *
135      * Target: https://teammatesv4.appspot.com
136      * Origin: https://teammatesv4.appspot.com
137      * </pre>
138      * Origin matches target. This request is valid.</p>
139      */
140     private boolean isHttpReferrerValid(String referrer, String requestUrl) {
141         String origin;
142         try {
143             origin = new Url(referrer).getBaseUrl();
144         } catch (AssertionError e) { // due to MalformedURLException
145             return false;
146         }
147 
148         if (Config.isDevServer() && origin.equals(Config.APP_FRONTENDDEV_URL)) {
149             // Exception to the rule: front-end dev server requesting data from back-end dev server
150             return true;
151         }
152 
153         String target = new Url(requestUrl).getBaseUrl();
154         return origin.replaceFirst("^https?://", "").equals(target.replaceFirst("^https?://", ""));
155     }

157     private String getCsrfTokenErrorIfAny(HttpServletRequest request) {
158         String csrfToken = request.getHeader(Const.HeaderNames.CSRF_TOKEN);
159         if (csrfToken == null || csrfToken.isEmpty()) {
160             return "Missing CSRF token.";
161         }
162 
163         String sessionId = request.getRequestedSessionId();
164         if (sessionId == null) {
165             // Newly-created session

171         } catch (InvalidParametersException e) {
