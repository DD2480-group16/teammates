180      * Schedules for the given list of emails to be sent.
181      *
182      * @param emails the list of emails to be sent
183      */
184     public void scheduleEmailsForSending(List<EmailWrapper> emails) {
185         if (emails.isEmpty()) {
186             return;
187         }
188 
189         // Equally spread out the emails to be sent over 1 hour
190         // Sets interval to a maximum of 5 seconds if the interval is too large
191         int oneHourInMillis = 60 * 60 * 1000;
192         int emailIntervalMillis = Math.min(5000, oneHourInMillis / emails.size());
193 
194         int numberOfEmailsSent = 0;
195         for (EmailWrapper email : emails) {
196             long emailDelayTimer = (long) numberOfEmailsSent * (long) emailIntervalMillis;
197             scheduleEmailForSending(email, emailDelayTimer);
198             numberOfEmailsSent++;
199         }
200     }
