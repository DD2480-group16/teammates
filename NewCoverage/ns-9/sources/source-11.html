156     private boolean isRequestFilterSatisfied(LogDetails details, String actionClassFilter,
157             String latencyFilter, String statusFilter, String regkeyFilter, String emailFilter, String googleIdFilter) {
158         if (actionClassFilter == null && latencyFilter == null && statusFilter == null
159                 && regkeyFilter == null && emailFilter == null && googleIdFilter == null) {
160             return true;
161         }
162         if (details.getEvent() != LogEvent.REQUEST_LOG) {
163             return false;
164         }
165         RequestLogDetails requestDetails = (RequestLogDetails) details;
166         if (actionClassFilter != null && !actionClassFilter.equals(requestDetails.getActionClass())) {
167             return false;
168         }
169         if (statusFilter != null && !statusFilter.equals(String.valueOf(requestDetails.getResponseStatus()))) {
170             return false;
171         }
172         if (latencyFilter != null) {
173             Pattern p = Pattern.compile("^(>|>=|<|<=) *(\\d+)$");
174             Matcher m = p.matcher(latencyFilter);
175             long logLatency = ((RequestLogDetails) details).getResponseTime();
176             boolean isFilterSatisfied = false;
177             if (m.matches()) {
178                 int time = Integer.parseInt(m.group(2));
179                 switch (m.group(1)) {
180                 case ">":
181                     isFilterSatisfied = logLatency > time;
182                     break;
183                 case ">=":
184                     isFilterSatisfied = logLatency >= time;
185                     break;
186                 case "<":
187                     isFilterSatisfied = logLatency < time;
188                     break;
189                 case "<=":
190                     isFilterSatisfied = logLatency <= time;
191                     break;
192                 default:
193                     assert false : "Unreachable case";
194                     break;
195                 }
196             }
197             if (!isFilterSatisfied) {
198                 return false;
199             }
200         }
201         RequestLogUser userInfo = requestDetails.getUserInfo();
202         if (regkeyFilter != null && (userInfo == null || !regkeyFilter.equals(userInfo.getRegkey()))) {
203             return false;
204         }
205         if (emailFilter != null && (userInfo == null || !emailFilter.equals(userInfo.getEmail()))) {
206             return false;
207         }
208         return googleIdFilter == null || (userInfo != null && googleIdFilter.equals(userInfo.getGoogleId()));
209     }
