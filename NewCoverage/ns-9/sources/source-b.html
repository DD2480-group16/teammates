54     public List<ErrorLogEntry> getRecentErrorLogs() {
55         Instant endTime = Instant.now();
56         // Sets the range to 6 minutes to slightly overlap the 5 minute email timer
57         long queryRange = 1000 * 60 * 6;
58         Instant startTime = endTime.minusMillis(queryRange);
59 
60         QueryLogsParams queryLogsParams = QueryLogsParams.builder(startTime.toEpochMilli(), endTime.toEpochMilli())
61                 .withMinSeverity(LogSeverity.ERROR)
62                 .build();
63         LogSearchParams logSearchParams = LogSearchParams.from(queryLogsParams)
64                 .addLogName(REQUEST_LOG_NAME)
65                 .setResourceType(RESOURCE_TYPE_GAE_APP);
66 
67         List<LogEntry> logEntries = new ArrayList<>();
68         List<ErrorLogEntry> errorLogs = new ArrayList<>();
69 
70         Page<LogEntry> entries = getLogEntries(logSearchParams, 0);
71         for (LogEntry entry : entries.iterateAll()) {
72             logEntries.add(entry);
73         }
74 
75         for (LogEntry logEntry : logEntries) {
76             Any entry = (Any) logEntry.getPayload().getData();
77 
78             JsonFormat.TypeRegistry tr = JsonFormat.TypeRegistry.newBuilder()
79                     .add(RequestLog.getDescriptor())
80                     .add(LogLine.getDescriptor())
81                     .add(com.google.appengine.logging.v1.SourceLocation.getDescriptor())
82                     .add(SourceReference.getDescriptor())
83                     .build();
84 
85             List<LogLine> logLines = new ArrayList<>();
86             try {
87                 String logContentAsJson = JsonFormat.printer().usingTypeRegistry(tr).print(entry);
88 
89                 RequestLog.Builder builder = RequestLog.newBuilder();
90                 JsonFormat.parser().ignoringUnknownFields().usingTypeRegistry(tr).merge(logContentAsJson, builder);
91                 RequestLog reconvertedLog = builder.build();
92 
93                 logLines = reconvertedLog.getLineList();
94             } catch (InvalidProtocolBufferException e) {
95                 // TODO
96             }
97 
98             String trace = logEntry.getTrace();
99             if (trace != null) {
100                 trace = trace.replace(TRACE_PREFIX, "");
101             }
102 
103             for (LogLine line : logLines) {
104                 if (line.getSeverity().getNumber() >= com.google.logging.type.LogSeverity.ERROR.getNumber()) {
105                     errorLogs.add(new ErrorLogEntry(
106                             line.getLogMessage().replaceAll("\n", "\n<br>"),
107                             line.getSeverity().toString(), trace)
108                     );
109                 }
110             }
111         }
112         return errorLogs;

124         List<GeneralLogEntry> logEntries = new ArrayList<>();

126             Severity severity = entry.getSeverity();
127             String trace = entry.getTrace();

129                 trace = trace.replace(TRACE_PREFIX, "");
130             }
131             String insertId = entry.getInsertId();
132             com.google.cloud.logging.SourceLocation sourceLocation = entry.getSourceLocation();
133             Map<String, String> resourceIdentifier = entry.getResource().getLabels();
134             Payload<?> payload = entry.getPayload();
135             long timestamp = entry.getTimestamp();
136 
137             String file = "";
138             Long line = 0L;
139             String function = "";
140             if (sourceLocation != null) {
141                 file = sourceLocation.getFile();
142                 line = sourceLocation.getLine();
143                 function = sourceLocation.getFunction();
144             }
145 
146             GeneralLogEntry logEntry = new GeneralLogEntry(convertSeverity(severity), trace, insertId,
147                     resourceIdentifier, new SourceLocation(file, line, function), timestamp);
148             if (payload.getType() == Payload.Type.JSON) {
149                 Map<String, Object> jsonPayloadMap = ((Payload.JsonPayload) payload).getDataAsMap();
150                 logEntry.setDetails(JsonUtils.fromJson(JsonUtils.toCompactJson(jsonPayloadMap), LogDetails.class));
151             } else {
152                 String textPayloadMessage = ((Payload.StringPayload) payload).getData();
153                 logEntry.setMessage(textPayloadMessage);
154             }
155             logEntries.add(logEntry);
156         }
157         boolean hasNextPage = logEntriesInPage.getNextPageToken() != null;
158         return new QueryLogsResults(logEntries, hasNextPage);

161     private LogSeverity convertSeverity(Severity severity) {
162         if (severity == Severity.ERROR) {
163             return LogSeverity.ERROR;
164         }
165         if (severity == Severity.WARNING) {
166             return LogSeverity.WARNING;
167         }
168         if (severity == Severity.INFO || severity == Severity.NOTICE) {
169             return LogSeverity.INFO;
170         }
171         if (severity == Severity.CRITICAL || severity == Severity.ALERT || severity == Severity.EMERGENCY) {
172             return LogSeverity.CRITICAL;
173         }
174         if (severity == Severity.DEBUG) {
175             return LogSeverity.DEBUG;
176         }
177         return LogSeverity.DEFAULT;

244         logFilters.add("timestamp<=\"" + Instant.ofEpochMilli(q.getEndTime()).toString() + "\"");
245 
246         if (!s.logName.isEmpty()) {
247             String logNameFilter = s.logName.stream()
248                     .map(str -> "\"projects/" + options.getProjectId() + "/logs/" + str + "\"")
249                     .collect(Collectors.joining(" OR "));

251         }
252         if (s.resourceType != null) {
253             logFilters.add("resource.type=\"" + s.resourceType + "\"");
254         }
255         if (q.getSeverity() != null) {
256             logFilters.add("severity=" + q.getSeverity());
257         } else if (q.getMinSeverity() != null && q.getSeverity() == null) {
258             logFilters.add("severity>=" + q.getMinSeverity());
259         }
260         if (q.getTraceId() != null) {
261             logFilters.add("trace=\"" + TRACE_PREFIX + q.getTraceId() + "\"");
262         }
263         if (q.getActionClass() != null) {
264             logFilters.add("jsonPayload.actionClass=\"" + q.getActionClass() + "\"");
265         }
266         if (q.getUserInfoParams() != null) {
267             if (q.getUserInfoParams().getGoogleId() != null) {
268                 logFilters.add("jsonPayload.userInfo.googleId=\"" + q.getUserInfoParams().getGoogleId() + "\"");
269             }

271                 logFilters.add("jsonPayload.userInfo.regkey=\"" + q.getUserInfoParams().getRegkey() + "\"");
272             }
273             if (q.getUserInfoParams().getEmail() != null) {
274                 logFilters.add("jsonPayload.userInfo.email=\"" + q.getUserInfoParams().getEmail() + "\"");
275             }
276         }
277         if (q.getLogEvent() != null) {
278             logFilters.add("jsonPayload.event=\"" + q.getLogEvent() + "\"");
279         }
280         if (q.getSourceLocation() != null && q.getSourceLocation().getFile() != null) {
281             if (q.getSourceLocation().getFunction() == null) {
282                 logFilters.add("sourceLocation.file=\"" + q.getSourceLocation().getFile() + "\"");
283             } else {

287         }
288         if (q.getExceptionClass() != null) {

290         }

292             logFilters.add("jsonPayload.responseTime" + q.getLatency());

324         try {
