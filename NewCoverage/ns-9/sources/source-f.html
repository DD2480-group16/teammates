178     public InstructorAttributes updateInstructorByGoogleIdCascade(

180             throws InstructorUpdateException, InvalidParametersException, EntityDoesNotExistException {
181 
182         InstructorAttributes originalInstructor =
183                 instructorsDb.getInstructorForGoogleId(updateOptions.getCourseId(), updateOptions.getGoogleId());
184 
185         if (originalInstructor == null) {
186             throw new EntityDoesNotExistException("Trying to update non-existent Entity: " + updateOptions);
187         }
188 
189         InstructorAttributes newInstructor = originalInstructor.getCopy();
190         newInstructor.update(updateOptions);
191

195 
196         InstructorAttributes updatedInstructor = instructorsDb.updateInstructorByGoogleId(updateOptions);
197 
198         if (!originalInstructor.getEmail().equals(updatedInstructor.getEmail())) {
199             // cascade responses

201                     frLogic.getFeedbackResponsesFromGiverForCourse(

203             for (FeedbackResponseAttributes responseFromUser : responsesFromUser) {

214                     }

335      * Checks if there are any other registered instructors that can modify instructors.
336      * If there are none, the instructor currently being edited will be granted the privilege
337      * of modifying instructors automatically.
338      *
339      * @param courseId         Id of the course.
340      * @param instructorToEdit Instructor that will be edited.
341      *                         This may be modified within the method.
342      */
343     public void updateToEnsureValidityOfInstructorsForTheCourse(String courseId, InstructorAttributes instructorToEdit) {
344         List<InstructorAttributes> instructors = getInstructorsForCourse(courseId);
345         int numOfInstrCanModifyInstructor = 0;
346         InstructorAttributes instrWithModifyInstructorPrivilege = null;
347         for (InstructorAttributes instructor : instructors) {
348             if (instructor.isAllowedForPrivilege(Const.InstructorPermissions.CAN_MODIFY_INSTRUCTOR)) {
349                 numOfInstrCanModifyInstructor++;
350                 instrWithModifyInstructorPrivilege = instructor;
351             }
352         }
353         boolean isLastRegInstructorWithPrivilege = numOfInstrCanModifyInstructor <= 1
354                 && instrWithModifyInstructorPrivilege != null
355                 && (!instrWithModifyInstructorPrivilege.isRegistered()
356                 || instrWithModifyInstructorPrivilege.getGoogleId()
357                 .equals(instructorToEdit.getGoogleId()));
358         if (isLastRegInstructorWithPrivilege) {
359             instructorToEdit.getPrivileges().updatePrivilege(Const.InstructorPermissions.CAN_MODIFY_INSTRUCTOR, true);
360         }
361     }
