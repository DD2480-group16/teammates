134     private <T extends EntityAttributes<?>> void updateDataBundleValue(List<T> newValues, Map<String, T> oldValues) {
135         Map<T, Integer> newValuesMap = new HashMap<>();
136         Map<String, T> values = new LinkedHashMap<>();
137 
138         for (int i = 0; i < newValues.size(); i++) {
139             newValuesMap.put(newValues.get(i), i);
140         }
141 
142         for (Map.Entry<String, T> entry : oldValues.entrySet()) {
143             String key = entry.getKey();
144             T value = entry.getValue();
145 
146             if (newValuesMap.containsKey(value)) {
147                 int index = newValuesMap.get(value);
148                 values.put(key, newValues.get(index));
149             }
150         }
151 
152         oldValues.clear();
153         oldValues.putAll(values);
154     }

233      * This method is necessary to generate the feedbackQuestionId of the
234      * question the response is for.<br>
235      * Normally, the ID is already generated on creation,
236      * but the json file does not contain the actual response ID. <br>
237      * Therefore the question number corresponding to the created response

242             Map<String, String> questionIdMap) {
243         for (FeedbackResponseAttributes response : responses) {
244             int questionNumber;

247             } catch (NumberFormatException e) {

289      * Checks if the role of {@code instructor} matches its privileges.
290      *
291      * @param instructor
292      *            the {@link InstructorAttributes} of an instructor, cannot be
293      *            {@code null}
294      */
295     private void validateInstructorPrivileges(InstructorAttributes instructor) {
296 
297         if (instructor.getRole() == null) {
298             return;
299         }
300 
301         InstructorPrivileges privileges = instructor.getPrivileges();
302 
303         switch (instructor.getRole()) {
304 
305         case Const.InstructorPermissionRoleNames.INSTRUCTOR_PERMISSION_ROLE_COOWNER:
306             assert privileges.hasCoownerPrivileges();
307             break;
308 
309         case Const.InstructorPermissionRoleNames.INSTRUCTOR_PERMISSION_ROLE_MANAGER:
310             assert privileges.hasManagerPrivileges();
311             break;
312 
313         case Const.InstructorPermissionRoleNames.INSTRUCTOR_PERMISSION_ROLE_OBSERVER:
314             assert privileges.hasObserverPrivileges();
315             break;
316 
317         case Const.InstructorPermissionRoleNames.INSTRUCTOR_PERMISSION_ROLE_TUTOR:
318             assert privileges.hasTutorPrivileges();
319             break;
320 
321         case Const.InstructorPermissionRoleNames.INSTRUCTOR_PERMISSION_ROLE_CUSTOM:
322             break;
323 
324         default:
325             assert false : "Invalid instructor permission role name";
326             break;
327         }
328     }
