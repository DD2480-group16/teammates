259      *
260      * <p>Adjust email sending status if necessary.
261      *
262      * @return updated feedback session
263      * @throws InvalidParametersException if attributes to update are not valid
264      * @throws EntityDoesNotExistException if the feedback session cannot be found
265      */
266     public FeedbackSessionAttributes updateFeedbackSession(FeedbackSessionAttributes.UpdateOptions updateOptions)
267             throws InvalidParametersException, EntityDoesNotExistException {
268         FeedbackSessionAttributes oldSession =
269                 fsDb.getFeedbackSession(updateOptions.getCourseId(), updateOptions.getFeedbackSessionName());
270 
271         if (oldSession == null) {
272             throw new EntityDoesNotExistException(ERROR_NON_EXISTENT_FS_UPDATE + updateOptions.getCourseId()
273                     + "/" + updateOptions.getFeedbackSessionName());
274         }
275 
276         FeedbackSessionAttributes newSession = oldSession.getCopy();
277         newSession.update(updateOptions);
278         FeedbackSessionAttributes.UpdateOptions.Builder newUpdateOptions =
279                 FeedbackSessionAttributes.updateOptionsBuilder(updateOptions);
280 
281         // adjust email sending status
282 
283         // reset sentOpenEmail if the session has opened but is being un-opened
284         // now, or else leave it as sent if so.
285         if (oldSession.isSentOpenEmail()) {
286             newUpdateOptions.withSentOpenEmail(newSession.isOpened());
287 
288             // also reset sentOpeningSoonEmail
289             newUpdateOptions.withSentOpeningSoonEmail(
290                     newSession.isOpened()
291                             || newSession.isOpeningInHours(NUMBER_OF_HOURS_BEFORE_OPENING_SOON_ALERT));
292         }
293 
294         // reset sentClosedEmail if the session has closed but is being un-closed
295         // now, or else leave it as sent if so.
296         if (oldSession.isSentClosedEmail()) {
297             newUpdateOptions.withSentClosedEmail(newSession.isClosed());
298 
299             // also reset sentClosingEmail
300             newUpdateOptions.withSentClosingEmail(
301                     newSession.isClosed()
302                             || newSession.isClosedAfter(NUMBER_OF_HOURS_BEFORE_CLOSING_ALERT));
303         }
304 
305         // reset sentPublishedEmail if the session has been published but is
306         // going to be unpublished now, or else leave it as sent if so.
307         if (oldSession.isSentPublishedEmail()) {
308             newUpdateOptions.withSentPublishedEmail(newSession.isPublished());
309         }
310 
311         return fsDb.updateFeedbackSession(newUpdateOptions.build());
312     }

566                     || isInstructor && frLogic.isResponseOfFeedbackQuestionVisibleToInstructor(question)) {
567                 // We only need one question with visible responses for the entire session to be visible
568                 questionsWithVisibleResponses.add(question);
569                 break;

571         }
572
